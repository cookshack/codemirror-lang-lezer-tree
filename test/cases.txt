# Lone

X

==>
Tree(TopNode(Name))

# Child

X(Y)

==>
Tree(TopNode(Name,
             Children(Node(Name))))

# Children

X(Y1, Y2)

==>
Tree(TopNode(Name,
             Children(Node(Name),
                      Node(Name))))

# Grandchildren

X(Y1, Y2(Z1, Z2), Y3(Z3))

==>
Tree(TopNode(Name,
             Children(Node(Name),
                      Node(Name,
                           Children(Node(Name),
                                    Node(Name))),
                      Node(Name,
                           Children(Node(Name))))))

# Special

X(Y1, keyword)

==>
Tree(TopNode(Name,
             Children(Node(Name),
                      Special)))
